  function calcB(E,T,i) {
      var tempbpk=0;
      var tempbk=0;
      for(var j=0;j< T.length;j++)
      {
          if(E[i].pib>T[j].pib)
          {
            tempbpk+=T[j].p;
          }
          else
          {
            tempbk+=T[j].b;
          }
      }
